Câu 4: Cho biết kết quả in ra màn hình của đoạn chương trình sau:
#include<iostream>
using namespace std;
void Swap(int &n,int a)
{ 
    int tam;
    tam = n;
    n = a;
    a = tam;
}
int main()
{ 
    int n=5;
    int *a= new int(1);
    Swap(n,*a);
    cout<<n<<" "<<*a;
    return 0;
}
Đáp án:
1 1.
1 5
5 1
5 5
